```typescript
import { createReadStream } from 'fs';
import { readFileSync } from 'fs';
import { resolve } from 'path';
import createGraph from 'ngraph.graph';
import { aStar } from 'ngraph.path';

// ─── Data Loading (once at module init) ───────────────────────────────────
const GRAPH_PATH = resolve(process.cwd(), '../public/data/graph.json');
const FRICTION_PATH = resolve(process.cwd(), '../public/data/friction_cache.json');

// Load or stub if Phase 2 artifacts not yet available
function loadJson<T>(path: string, fallback: T): T {
  try {
    return JSON.parse(readFileSync(path, 'utf-8')) as T;
  } catch {
    console.warn(`[astar] Could not load ${path} — using fallback`);
    return fallback;
  }
}

interface GraphNode { id: string; lat: number; lng: number; regulatoryRisk?: number; }
interface GraphEdge { from: string; to: string; frictionScore: number; distanceKm: number; regulatoryRisk?: number; }
interface GraphJson { nodes: GraphNode[]; edges: GraphEdge[]; }
interface FrictionNode { lat: number; lng: number; frictionScore: number; justification: string; }
type FrictionCache = Record<string, FrictionNode>;

const STUB_GRAPH: GraphJson = { nodes: [], edges: [] };
const graphJson = loadJson<GraphJson>(GRAPH_PATH, STUB_GRAPH);
const frictionCache = loadJson<FrictionCache>(FRICTION_PATH, {});

// ─── Haversine heuristic ───────────────────────────────────────────────────
function haversineKm(lat1: number, lng1: number, lat2: number, lng2: number): number {
  const R = 6371;
  const dLat = ((lat2 - lat1) * Math.PI) / 180;
  const dLng = ((lng2 - lng1) * Math.PI) / 180;
  const a =
    Math.sin(dLat / 2) ** 2 +
    Math.cos((lat1 * Math.PI) / 180) * Math.cos((lat2 * Math.PI) / 180) * Math.sin(dLng / 2) ** 2;
  return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
}

// ─── Graph construction ───────────────────────────────────────────────────
export function buildGraph(json: GraphJson = graphJson) {
  const g = createGraph<GraphNode, GraphEdge>();
  for (const node of json.nodes) {
    g.addNode(node.id, node);
  }
  for (const edge of json.edges) {
    // Augment edge with frictionScore from cache if edge doesn't include it
    const cachedFriction = frictionCache[edge.from]?.frictionScore ?? edge.frictionScore ?? 0.5;
    const cachedRisk = frictionCache[edge.from]?.frictionScore ?? edge.regulatoryRisk ?? 0.5;
    g.addLink(edge.from, edge.to, { ...edge, frictionScore: cachedFriction, regulatoryRisk: cachedRisk });
  }
  return g;
}

export const sharedGraph = buildGraph();

// ─── Nearest node lookup ──────────────────────────────────────────────────
export function findNearestNode(lat: number, lng: number, nodes: GraphNode[] = graphJson.nodes): string {
  let nearest = nodes[0]?.id ?? 'node-0';
  let minDist = Infinity;
  for (const node of nodes) {
    const dist = (node.lat - lat) ** 2 + (node.lng - lng) ** 2;
    if (dist < minDist) { minDist = dist; nearest = node.id; }
  }
  return nearest;
}

// ─── Route finding ────────────────────────────────────────────────────────
export interface RouteWeights {
  costW: number;
  riskW: number;
  coLocationW: number;
}

export function findRoute(
  fromNodeId: string,
  toNodeId: string,
  graph = sharedGraph,
  weights: RouteWeights = { costW: 1.0, riskW: 1.0, coLocationW: 1.0 },
): string[] {
  const finder = aStar(graph, {
    distance(fromNode, toNode, link) {
      const data = link.data as GraphEdge;
      return (data.frictionScore ?? 0.5) * weights.costW + (data.regulatoryRisk ?? 0.5) * weights.riskW;
    },
    heuristic(from, to) {
      return haversineKm(
        (from.data as GraphNode).lat,
        (from.data as GraphNode).lng,
        (to.data as GraphNode).lat,
        (to.data as GraphNode).lng,
      );
    },
  });

  const path = finder.find(fromNodeId, toNodeId);
  return path.map((n) => n.id);
}

// ─── Exports for route handler ────────────────────────────────────────────
export { graphJson, frictionCache };
```

```typescript
import { Router } from 'express';
import { findNearestNode, findRoute, sharedGraph, graphJson, frictionCache } from '../routing/astar.js';

const router = Router();

// Constraint slider: 0=full cost, 1=full risk; blends into profile base weights
function blendWeights(base: { costW: number; riskW: number }, costRisk: number) {
  const blend = Math.max(0, Math.min(1, costRisk));
  return {
    costW: base.costW * (1 - blend * 0.5),
    riskW: base.riskW * (1 + blend * 0.5),
    coLocationW: 1.0,
  };
}

function nodeIdPathToLineString(nodeIds: string[]) {
  const coords = nodeIds.map((id) => {
    const node = graphJson.nodes.find((n) => n.id === id);
    return node ? [node.lng, node.lat] : [0, 0];
  });
  return { type: 'LineString' as const, coordinates: coords };
}

function buildSegmentJustifications(nodeIds: string[]) {
  return nodeIds.slice(0, -1).map((id, i) => ({
    segmentIndex: i,
    frictionScore: frictionCache[id]?.frictionScore ?? 0.5,
    justification: frictionCache[id]?.justification ?? 'No justification data available.',
  }));
}

function haversineKm(lat1: number, lng1: number, lat2: number, lng2: number) {
  const R = 6371;
  const dLat = ((lat2 - lat1) * Math.PI) / 180;
  const dLng = ((lng2 - lng1) * Math.PI) / 180;
  const a = Math.sin(dLat / 2) ** 2 +
    Math.cos((lat1 * Math.PI) / 180) * Math.cos((lat2 * Math.PI) / 180) * Math.sin(dLng / 2) ** 2;
  return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
}

function totalDistanceMiles(nodeIds: string[]) {
  let totalKm = 0;
  for (let i = 0; i < nodeIds.length - 1; i++) {
    const a = graphJson.nodes.find((n) => n.id === nodeIds[i]);
    const b = graphJson.nodes.find((n) => n.id === nodeIds[i + 1]);
    if (a && b) totalKm += haversineKm(a.lat, a.lng, b.lat, b.lng);
  }
  return totalKm * 0.621371;
}

// Stub response when graph is empty (Phase 2 incomplete)
function cannedStubRoutes(source: [number, number], dest: [number, number]) {
  const geo = (a: [number, number], b: [number, number]) => ({
    type: 'LineString' as const,
    coordinates: [a, b],
  });
  return [
    { id: 'A', profile: 'lowest-cost', label: 'Route A — Lowest Cost', color: '#A7C8FF',
      geometry: geo(source, dest), metrics: { distanceMiles: 120, estimatedCapexUSD: 420_000_000, permittingMonths: [18, 24] as [number,number] },
      segmentJustifications: [{ segmentIndex: 0, frictionScore: 0.3, justification: 'Low-friction corridor along US-385 in Reeves County.' }], narrativeSummary: '' },
    { id: 'B', profile: 'balanced', label: 'Route B — Balanced', color: '#FFBC7C',
      geometry: geo(source, dest), metrics: { distanceMiles: 135, estimatedCapexUSD: 567_000_000, permittingMonths: [24, 36] as [number,number] },
      segmentJustifications: [{ segmentIndex: 0, frictionScore: 0.5, justification: 'Mixed terrain through Edwards Plateau.' }], narrativeSummary: '' },
    { id: 'C', profile: 'lowest-risk', label: 'Route C — Lowest Regulatory Risk', color: '#E8B3FF',
      geometry: geo(source, dest), metrics: { distanceMiles: 155, estimatedCapexUSD: 775_000_000, permittingMonths: [30, 48] as [number,number] },
      segmentJustifications: [{ segmentIndex: 0, frictionScore: 0.2, justification: 'Avoids Edwards Aquifer recharge zone and Nolan County habitat clusters.' }], narrativeSummary: '' },
  ];
}

router.post('/route', async (req, res) => {
  try {
    const { source, dest, constraints } = req.body as {
      source: [number, number];
      dest: [number, number];
      constraints: { costRisk: number; coLocation: boolean; eminentDomainAvoidance: boolean; ecologyAvoidance: boolean };
    };

    const costRisk = constraints?.costRisk ?? 0.5;

    // If graph has no nodes, return stub routes so UI isn't blocked
    if (graphJson.nodes.length === 0) {
      return res.json(cannedStubRoutes(source, dest));
    }

    const srcId = findNearestNode(source[1], source[0]); // lat, lng order
    const dstId = findNearestNode(dest[1], dest[0]);

    const [pathA, pathB, pathC] = await Promise.all([
      Promise.resolve(findRoute(srcId, dstId, sharedGraph, blendWeights({ costW: 1.5, riskW: 0.5 }, costRisk))),
      Promise.resolve(findRoute(srcId, dstId, sharedGraph, blendWeights({ costW: 1.0, riskW: 1.0 }, costRisk))),
      Promise.resolve(findRoute(srcId, dstId, sharedGraph, blendWeights({ costW: 0.5, riskW: 1.5 }, costRisk))),
    ]);

    const routes = [
      { id: 'A', profile: 'lowest-cost', label: 'Route A — Lowest Cost', color: '#A7C8FF', path: pathA, capexPerMile: 3_500_000, permitting: [18, 24] as [number, number] },
      { id: 'B', profile: 'balanced',    label: 'Route B — Balanced',     color: '#FFBC7C', path: pathB, capexPerMile: 4_200_000, permitting: [24, 36] as [number, number] },
      { id: 'C', profile: 'lowest-risk', label: 'Route C — Lowest Regulatory Risk', color: '#E8B3FF', path: pathC, capexPerMile: 5_000_000, permitting: [30, 48] as [number, number] },
    ].map((r) => {
      const miles = totalDistanceMiles(r.path);
      return {
        id: r.id,
        profile: r.profile,
        label: r.label,
        color: r.color,
        geometry: nodeIdPathToLineString(r.path),
        metrics: {
          distanceMiles: Math.round(miles),
          estimatedCapexUSD: Math.round(miles * r.capexPerMile),
          permittingMonths: r.permitting,
        },
        segmentJustifications: buildSegmentJustifications(r.path),
        narrativeSummary: '',
      };
    });

    res.json(routes);
  } catch (err) {
    console.error('[POST /api/route]', err);
    res.status(500).json({ error: 'Route generation failed' });
  }
});

export default router;
```
