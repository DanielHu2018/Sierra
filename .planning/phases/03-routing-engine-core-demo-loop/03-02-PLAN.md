```typescript
// ─── Phase 3: New type contracts ───────────────────────────────────────────

export interface RouteRecommendation {
  routeId: 'A' | 'B' | 'C';
  rationale: string;       // LLM-generated 3-sentence rationale
  timestamp: number;       // Unix ms — for cache invalidation
}

export interface TriggerEntry {
  statute: string;                    // e.g. "ESA Section 7"
  explanation: string;                // plain-English
  timelineMonths: [number, number];   // [min, max] contribution
}

export interface EnvironmentalTrigger {
  routeId: 'A' | 'B' | 'C';
  triggers: TriggerEntry[];
}

export interface AlertItem {
  text: string;
  location: string;    // Texas location name for credibility
}

export interface SierraAlert {
  primary: AlertItem;
  secondary: AlertItem[];  // up to 2 collapsed secondary alerts
}

export interface ProjectPhase {
  name: string;                        // e.g. "Desktop Screening"
  estimatedMonths: [number, number];   // [min, max]
  keyDependency: string;
}

export interface ProjectSummary {
  phases: ProjectPhase[];
  // Six canonical phases: Desktop Screening, Environmental Review,
  // ROW Acquisition, State Permitting, Construction, Total Estimated Timeline
}

export interface FrictionNode {
  lat: number;
  lng: number;
  frictionScore: number;   // 0–1
  justification: string;
}

export type FrictionCache = Record<string, FrictionNode>;
```

Note: `FrictionCache` is added here because it is used by both the heatmap (OverlayLayers) and hover justifications (RouteLayer), and needs a shared type definition.
