```typescript
import OpenAI from 'openai';
const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

async function embedChunks(texts: string[]): Promise<number[][]> {
  const response = await openai.embeddings.create({
    model: 'text-embedding-3-small',
    input: texts,
    encoding_format: 'float',
  });
  return response.data.map(d => d.embedding);
}
```

**Chunking strategy (Claude's discretion — paragraph-level):**
Split each statute text by `\n\n` (paragraph breaks) or every 400 tokens if paragraphs are too long. Target 5–10 chunks per statute. Filter out chunks shorter than 50 characters.

**Two-path fetch + fallback:**
```typescript
async function fetchStatuteText(statute: string, url: string): Promise<string> {
  try {
    const res = await fetch(url, {
      headers: { 'User-Agent': 'Mozilla/5.0 (Sierra Pipeline)' },
      signal: AbortSignal.timeout(10000),
    });
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    const contentType = res.headers.get('content-type') ?? '';
    if (contentType.includes('text/html')) {
      // HTML page — extract visible text (strip tags)
      const html = await res.text();
      const text = html.replace(/<[^>]+>/g, ' ').replace(/\s+/g, ' ').trim();
      if (text.length > 200) return text.slice(0, 8000); // cap at 8k chars
    }
    // PDF or plain text — return as-is (PDFs will be mostly binary, use fallback)
    throw new Error('PDF fetch not supported; using fallback');
  } catch (err) {
    console.warn(`[${statute}] Fetch failed (${(err as Error).message}), using fallback text`);
    return REGULATION_FALLBACKS[statute] ?? '';
  }
}
```
